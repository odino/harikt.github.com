195:         $this->results = clone $this->result_collection;

198:         $args = func_get_args();
199:         array_shift($args);
200:         array_shift($args);

207:     /**
208:      * 
209:      * Invokes the Handler objects for a sender and signal.
210:      * 
211:      * @param object $origin The object sending the signal. Note that this is
212:      * always an object, not a class name.
213:      * 
214:      * @param string $signal The name of the signal from that origin.
215:      * 
216:      * @param array $args Arguments to pass to the Handler callback.
217:      * 
218:      */
219:     protected function process($origin, $signal, array $args)
220:     {
221:         // are there any handlers for this signal, regardless of sender?
222:         $list = $this->getHandlers($signal);
223:         if (! $list) {
224:             return;
225:         }
226: 
227:         // go through the handler positions for the signal
228:         foreach ($list as $position => $handlers) {
229: 
230:             // go through each handler in this position
231:             foreach ($handlers as $handler) {
232: 
233:                 // try the handler
234:                 $params = $handler->exec($origin, $signal, $args);
235: 
236:                 // if it executed, it returned the params for a Result object
237:                 if ($params) {
238: 
239:                     // create a Result object
240:                     $result = $this->result_factory->newInstance($params);
241: 
242:                     // allow a meta-handler to examine the Result object,
243:                     // but only if it wasn't sent from the Manager (this
244:                     // prevents infinite looping). use process() instead
245:                     // of send() to prevent resetting the $results prop.
246:                     if ($origin !== $this) {
247:                         $this->process($this, 'handler_result', [$result]);
248:                     }
249: 
250:                     // retain the result
251:                     $this->results->append($result);
252: 
253:                     // should we stop processing?
254:                     if ($result->value === static::STOP) {
255:                         // yes, leave the processing loop
256:                         return;
257:                     }
258:                 }
259:             }
260:         }
261:     }
